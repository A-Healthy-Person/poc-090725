function handleBackButton() { // Check if we're in provider connection if (document.getElementById('provider-connect').style.display === 'block') { // Go back to FastenHealth search document.getElementById('provider-connect').style.display = 'none'; document.getElementById('provider-connect').classList.remove('active'); document.getElementById('fastenhealth-connect').style.display = 'block'; document.getElementById('fastenhealth-connect').classList.add('active'); return; } // Check if we're in FastenHealth connection if (document.getElementById('fastenhealth-connect').style.display === 'block') { // Go back to data sources document.getElementById('fastenhealth-connect').style.display = 'none'; document.getElementById('fastenhealth-connect').classList.remove('active'); document.getElementById('data-sources').style.display = 'block'; document.getElementById('data-sources').classList.add('active'); return; } // Check if we're in AWS setup if (document.getElementById('aws-setup').style.display === 'block') { // Go back to sovereignty selection document.getElementById('aws-setup').style.display = 'none'; document.getElementById('aws-setup').classList.remove('active'); document.getElementById('step-1').classList.add('active'); return; } // Check if we're in data sources if (document.getElementById('data-sources').style.display === 'block') { // Go back to vault login document.getElementById('data-sources').style.display = 'none'; document.getElementById('data-sources').classList.remove('active'); document.getElementById('step-2').classList.add('active'); return; } // Default: go back to landing page backToLanding(); }
